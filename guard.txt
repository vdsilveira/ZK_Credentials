webpack: (config, { isServer }) => {

  config.experiments = { asyncWebAssembly: true };

 

  if (isServer) {

    config.output.webassemblyModuleFilename =

      './../static/wasm/[modulehash].wasm';

  } else {

    config.output.webassemblyModuleFilename =

      'static/wasm/[modulehash].wasm';

  }

 

  return config;

},
